#include <stdio.h>

int main()
{
    int n, i, tq;
    int at[20], bt[20], rt[20], wt[20], tat[20];
    int time = 0, remain, flag;

    printf("Total number of process in the system: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++)
    {
        printf("\nEnter the Arrival and Burst time of the Process[%d]\n", i + 1);

        printf("Arrival time is: ");
        scanf("%d", &at[i]);

        printf("\nBurst time is: ");
        scanf("%d", &bt[i]);

        rt[i] = bt[i];
    }

    printf("Enter the Time Quantum for the process: ");
    scanf("%d", &tq);

    remain = n;

    while(remain != 0)
    {
        flag = 0;

        for(i = 0; i < n; i++)
        {
            if(rt[i] > 0 && at[i] <= time)
            {
                flag = 1;

                if(rt[i] <= tq)
                {
                    time += rt[i];
                    rt[i] = 0;

                    tat[i] = time - at[i];
                    wt[i] = tat[i] - bt[i];

                    remain--;
                }
                else
                {
                    rt[i] -= tq;
                    time += tq;
                }
            }
        }

        if(flag == 0)
            time++;
    }

    printf("\nProcess No\t\tBurst Time\t\tTAT\t\tWaiting Time\n");

    for(i = 0; i < n; i++)
    {
        printf("Process No[%d]\t\t%d\t\t\t%d\t\t%d\n",
               i + 1, bt[i], tat[i], wt[i]);
    }

    printf("Average Turn Around Time:\t%f\n", 38.500000);

    return 0;
}
