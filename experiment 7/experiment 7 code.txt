#include <stdio.h>

int main()
{
    int n, i, j, temp;
    int at[20], bt[20], p[20];
    int st[20], et[20], wt[20], tat[20];
    float avgwt = 0, avgtat = 0;

    printf("Enter the number of processes\n");
    scanf("%d", &n);

    for(i = 0; i < n; i++)
    {
        printf("Enter the arrival time and execution time for process %d\n", i + 1);
        scanf("%d%d", &at[i], &bt[i]);
        p[i] = i + 1;
    }

    for(i = 0; i < n - 1; i++)
    {
        for(j = i + 1; j < n; j++)
        {
            if(bt[i] > bt[j])
            {
                temp = bt[i]; bt[i] = bt[j]; bt[j] = temp;
                temp = at[i]; at[i] = at[j]; at[j] = temp;
                temp = p[i];  p[i] = p[j];   p[j] = temp;
            }
        }
    }

    st[0] = 0;
    et[0] = st[0] + bt[0];

    for(i = 1; i < n; i++)
    {
        st[i] = et[i - 1];
        et[i] = st[i] + bt[i];
    }

    for(i = 0; i < n; i++)
    {
        wt[i] = st[i] - at[i];
        tat[i] = et[i] - at[i];

        avgwt += wt[i];
        avgtat += tat[i];
    }

    printf("\nProcess |Arrival time  |Execution time |Start time    |End time      |Waiting\n");

    for(i = 0; i < n; i++)
    {
        printf("p[%d]   |\t%d\t|\t%d\t|\t%d\t|\t%d\t|\t%d\t|\t%d\n",
               p[i], at[i], bt[i], st[i], et[i], wt[i], tat[i]);
    }

    printf("Average waiting time is %f\n", avgwt / n);
    printf("Average turnaround time is %f\n", avgtat / n);

    return 0;
}
