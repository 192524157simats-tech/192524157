#include <stdio.h>

int main()
{
    int bt[20], p[20], wt[20], tat[20];
    int i, j, n, temp;
    float avgwt = 0, avgtat = 0;

    printf("Enter the total number of Processes: ");
    scanf("%d", &n);

    printf("\nPlease Enter the Burst Time and Priority of each process:\n");

    for(i = 0; i < n; i++)
    {
        printf("\nEnter the details of the process %c\n", 65 + i);

        printf("Enter the burst time: ");
        scanf("%d", &bt[i]);

        printf("Enter the priority: ");
        scanf("%d", &p[i]);
    }

    for(i = 0; i < n - 1; i++)
    {
        for(j = i + 1; j < n; j++)
        {
            if(p[i] < p[j])
            {
                temp = p[i];
                p[i] = p[j];
                p[j] = temp;

                temp = bt[i];
                bt[i] = bt[j];
                bt[j] = temp;
            }
        }
    }

    wt[0] = 0;

    for(i = 1; i < n; i++)
    {
        wt[i] = wt[i - 1] + bt[i - 1];
    }

    for(i = 0; i < n; i++)
    {
        tat[i] = wt[i] + bt[i];

        avgwt += wt[i];
        avgtat += tat[i];
    }

    printf("\n\nProcess_name\tBurst Time\tWaiting Time\tTurnaround Time\n");
    printf("=============================================================\n");

    for(i = 0; i < n; i++)
    {
        printf("\t%c\t\t%d\t\t%d\t\t%d\n",
               66 + i, bt[i], wt[i], tat[i]);

        printf("=============================================================\n");
    }

    printf("\nAverage Waiting Time : %f", avgwt / n);
    printf("\nAverage Turnaround Time: %f\n", avgtat / n);

    return 0;
}
