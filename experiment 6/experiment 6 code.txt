#include <stdio.h>

int main()
{
    int i, n, tq;
    int bt[20], rt[20];
    int wt = 0, tat = 0;
    int time = 0, remain;

    printf("Total number of process in the system: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++)
    {
        printf("\nEnter the Arrival and Burst time of the Process[%d]\n", i + 1);

        printf("Arrival time is: ");
        printf("0\n");

        printf("Burst time is: ");
        scanf("%d", &bt[i]);

        rt[i] = bt[i];
    }

    printf("\nEnter the Time Quantum for the process: ");
    scanf("%d", &tq);

    remain = n;

    while(remain != 0)
    {
        for(i = 0; i < n; i++)
        {
            if(rt[i] > 0)
            {
                if(rt[i] <= tq)
                {
                    time += rt[i];

                    tat += time;

                    wt += time - bt[i];

                    rt[i] = 0;

                    remain--;
                }
                else
                {
                    rt[i] -= tq;

                    time += tq;
                }
            }
        }
    }

    printf("\nProcess No\tBurst Time\tTAT\tWaiting Time\n");

    for(i = 0; i < n; i++)
    {
        printf("Process No[%d]\t%d\t\t%d\t%d\n",
               i + 1,
               bt[i],
               bt[i] + (wt / n),
               (wt / n));
    }

    printf("\nAverage Turn Around Time: %f", (float)tat / n);
    printf("\nAverage Waiting Time: %f\n", (float)wt / n);

    return 0;
}
